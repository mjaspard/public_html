CREATE TABLE BEAURAING(    id INT unsigned PRIMARY KEY NOT NULL AUTO_INCREMENT,    N VARCHAR(10),    NNE VARCHAR(10),    NE VARCHAR(10),    ENE VARCHAR(10),    E VARCHAR(10),    ESE VARCHAR(10),    SE VARCHAR(10),    SSE VARCHAR(10),    S VARCHAR(10),    SSO VARCHAR(10),    SO VARCHAR(10),    OSO VARCHAR(10),    O VARCHAR(10),    ONO VARCHAR(10),    NO VARCHAR(10),    NNO VARCHAR(10),    speed_max VARCHAR(10),    speed_min VARCHAR(10),    speed_avg VARCHAR(10),    sensor_avg int(11),    voltage_input double,    voltage_solar double,    voltage_battery double,    reading_time timestamp on update CURRENT_TIMESTAMP )
